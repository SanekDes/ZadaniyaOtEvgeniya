Числовое поле

Это то же простое тектовое поле, но с типом number. Тип из арсенала HTML5 позволяет огранить ввод данных только числами.

  //цифовое поле


  woocommerce_wp_text_input(  array(


 'id'			=>'_number_field',


 'label'		=>__( 'Цифровое поле', 'woocommerce' ),


 'placeholder'		=>'Ввод чисел',


 'description'		=>__( 'Вводятся только числа','woomcommerce'),


 'type'			=>'number',


 'custom_attribute'	=>  array(


	 'step'	=> 'any',


	 'min'	=> '0',


	),


) );
